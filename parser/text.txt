var 
	a:integer;
	vysl:integer;
function factorial(n:integer):integer;
var
	temp_result:integer;
	decremented_n:integer;
begin
	write('volani reku');
	if n<2 then begin
		factorial:=1
	end
	else begin
		decremented_n:=n-1;
		temp_result:=factorial(decremented_n);
		write('temp je: ', temp_result, ''#10'');
		factorial:=n * temp_result
	end
	end;
begin
	write('zadejte cislo pro vzpocet faktorialu: ');
	readln(a);
	if a<0 then begin
		write('faktorial nelze spocitat'#10'')
	end
	else begin
		vysl:=factorial(a);
		write('vysledek je: ', vysl, ''#10'')
	end
end.
